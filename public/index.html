Buy Now MFE Demo

React-based Micro Frontend for Buy Now button with product data logging and API validation.

Sample Product

$29.99

This is a sample product for testing the Buy Now MFE integration.

Buy Now Button

Integration

// Load the MFE
<script src="buy-now-mfe.js"></script>

// Initialize
BuyNowMFE.init({
    containerId: 'buy-now-container',
    config: {
        productId: 'PROD-001',
        productName: 'Sample Product',
        productPrice: 29.99,
        productCurrency: 'USD',
        color: '#FF3B30',
        size: 'medium',
        label: 'Buy Now',
        enabled: true,
        showPrice: true,
        apiEndpoint: 'https://api.example.com/purchase'
    },
    onBuyNow: function(purchaseData) {
        console.log('Purchase data:', purchaseData);
    },
    onValidation: async function(validationData) {
        console.log('Validation data:', validationData);
        return true;
    }
});
